from js import document, GlslRenderer, getData from pyodide.ffi import to_js import numpy as np import io # Load the data file_handler = io.BytesIO(getData("3d_sandboxes/sdf/mwb3D01.npy").to_py()) data = np.load(file_handler, allow_pickle=True).item() Mwb = data['Mwb'] # Start the volumetric rendering renderer_volumetric = GlslRenderer.new(document.querySelector("#canvas-volumetric")) renderer_volumetric.setup( document.querySelector("#fragment-shader-volumetric").text, to_js({ "nbFct": data["nbFct"], "nbSeg": data["nbSeg"], }) ); renderer_volumetric.setMatrix('u_Mwb', to_js(Mwb.flatten()), Mwb.shape[0] * Mwb.shape[1] * Mwb.shape[2], Mwb.shape[3], True) # Start the planar rendering renderer_planar = GlslRenderer.new(document.querySelector("#canvas-planar")) planeDistance = 0.5 renderer_planar.setUniform("u_planeDistance", planeDistance); renderer_planar.setup( document.querySelector("#fragment-shader-planar").text, to_js({ "nbFct": data["nbFct"], "nbSeg": data["nbSeg"], }) ); renderer_planar.setMatrix("u_Mwb", to_js(Mwb.flatten()), Mwb.shape[0] * Mwb.shape[1] * Mwb.shape[2], Mwb.shape[3], True) def onPlaneDistanceChanged(evt): global planeDistance if (evt.buttons == 1) and (evt.button == 0) and evt.shiftKey: planeDistance += -evt.movementX * 0.001 planeDistance = max(0.4, min(planeDistance, 0.728)) renderer_planar.setUniform("u_planeDistance", planeDistance); renderer_planar.canvas.onmousemove = onPlaneDistanceChanged;
