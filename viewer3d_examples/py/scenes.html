Scenes example (Python version)

This example allows to switch between different scenes. Some contain a robot that can be manipulated, others are examples from MuJoCo (to check the rendering), some others are used to test the implementation.

Note that due to the differences in the way MuJoCo and three.js are defining materials and lights, colors are usually brighter that in the MuJoCo visualization tool. You might need to adjust the lights and materials in your scene to obtain the rendering you want.


from viewer3d import Viewer3D, configs from js import document from pyodide.ffi import create_proxy document.getElementById('loader-container').style.display = 'none'; # Function to load the scene selected by the user def loadSelectedScene(evt=None): # Load the scene and retrieve the robot (if any) viewer3D.loadScene('/scenes/' + selector.value) if selector.value.find('panda') >= 0: robot = viewer3D.createRobot('panda', configs.Panda.new()) if robot is None: i = 1 while True: robot = viewer3D.createRobot(f'panda{i}', configs.Panda.new(), f'panda{i}_') if robot is None: robot = viewer3D.createRobot(f'panda{i}', configs.PandaNoHand.new(), f'panda{i}_') if robot is None: break i += 1 elif selector.value.find('ur5') >= 0: if selector.value.find('gripper') >= 0: viewer3D.createRobot('ur5', configs.UR5WithGripper.new()) else: viewer3D.createRobot('ur5', configs.UR5.new()) # Start the physics simulation viewer3D.physicsSimulatorPaused = False # Create the Viewer3D viewer3D = Viewer3D(document.getElementById('viewer3d')) # Event handling to change the scene when the user selects one selector = document.getElementById('scenes') selector.addEventListener('change', create_proxy(loadSelectedScene)) loadSelectedScene()
