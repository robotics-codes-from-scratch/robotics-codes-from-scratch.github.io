Points example (Python version)

This example shows how to add points to the scene.

  • The red point turns around the robot
  • The yellow point follows the end-effector projection on the XY plane, and has a label
from viewer3d import Viewer3D, configs from js import document from scipy.spatial.transform import Rotation as R import math document.getElementById('loader-container').style.display = 'none'; # This function will be called once per frame def update(delta, time): # Move the first point point1.position = [ math.cos(0.2 * math.pi * time), -math.sin(0.2 * math.pi * time), 0.7, ] # The second arrow follows the end-effector projection on the XY plane ee_transforms = robot.endEffectorTransforms point2.position = [ ee_transforms[0], ee_transforms[1], 0.0, ] # Create the Viewer3D viewer3D = Viewer3D(document.getElementById('viewer3d')) viewer3D.setRenderingCallback(update) # Load the scene and retrieve the robot viewer3D.loadScene('/scenes/panda.xml') robot = viewer3D.createRobot('panda', configs.Panda.new()) # Start the physics simulation viewer3D.physicsSimulatorPaused = False # Add some points point1 = viewer3D.addPoint( 'point1', [1.0, 0.0, 0.7], radius=0.01, color='#ff0000', shading=True, ) point2 = viewer3D.addPoint( 'point2', [0.0, 0.0, 0.0], radius=0.02, color='#cccc00', label='(f_1, f_2)', shading=True, )
