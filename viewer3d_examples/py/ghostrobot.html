Ghost robot example (Python version)

This example shows how to display a "ghost" robot, a robot only there to show a specific pose without affecting the scene.

from viewer3d import Viewer3D, Layers, configs from js import document document.getElementById('loader-container').style.display = 'none'; # Create the Viewer3D viewer3D = Viewer3D(document.getElementById('viewer3d')) # Load the scene and retrieve the robota viewer3D.loadScene('/scenes/ghost_panda.xml') robot = viewer3D.createRobot('panda', configs.Panda.new(), 'panda_') viewer3D.activateLayer(Layers.NoShadows); ghost = viewer3D.createRobot( 'ghost', configs.Panda.new(), 'ghost_', { 'use_light_toon_shader': True, 'color': [1.0, 0.5, 0.8], 'controlsEnabled': False, } ) pos = ghost.jointPositions pos[0] = -1.5 ghost.jointPositions = pos # Start the physics simulation viewer3D.physicsSimulatorPaused = False
