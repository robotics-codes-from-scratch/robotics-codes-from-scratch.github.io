Gaussians example (Python version)

This example shows how to add gaussians to the scene.

from viewer3d import Viewer3D, configs, sigmaFromQuaternionAndScale from js import three from js import document document.getElementById('loader-container').style.display = 'none'; # Create the Viewer3D viewer3D = Viewer3D(document.getElementById('viewer3d')) # Load the scene and retrieve the robot viewer3D.loadScene('/scenes/panda.xml') robot = viewer3D.createRobot('panda', configs.Panda.new()) # Start the physics simulation viewer3D.physicsSimulatorPaused = False # Add some gaussians sigma = sigmaFromQuaternionAndScale( three.Quaternion.new().setFromAxisAngle(three.Vector3.new(0.5, 1.0, 0.3).normalize(), 0.4), [0.1, 0.3, 0.2] ) gaussian1 = viewer3D.addGaussian( 'gaussian1', [0.5, 0.0, 1.0], sigma, '#ff0000' ); sigma = sigmaFromQuaternionAndScale( three.Quaternion.new().setFromAxisAngle(three.Vector3.new(2.0, 0.3, 1.0).normalize(), 0.4), [0.2, 0.5, 0.2] ) gaussian2 = viewer3D.addGaussian( 'gaussian2', [-0.1, 0.0, 0.5], sigma, '#ffff00' ); sigma = sigmaFromQuaternionAndScale( three.Quaternion.new().setFromAxisAngle(three.Vector3.new(0.0, 0.0, 1.0).normalize(), 1.5), [0.2, 0.3, 0.3] ) gaussian3 = viewer3D.addGaussian( 'gaussian3', [0.5, 0.7, 0.5], sigma, '#0000ff' );
