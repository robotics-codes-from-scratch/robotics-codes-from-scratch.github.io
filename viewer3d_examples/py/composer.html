Composer example (Python version)

This example shows how to use several renderering layers and render passes.

3 layers are used here:

  • Layer #0 (Base): Contains the scene (robot, floor and orange box)
  • Layer #1 (NoShadows): Contains the targets
  • Layer #2 (Top): Contains the red point, the depth buffer is cleared before rendering (so anything in that layer appears on top of objects in already-rendered layers)

Additionaly, an OutlinePass is used to highlight the robot.

from viewer3d import Viewer3D, Shapes, Layers, Passes, OutlinePass, configs from js import document from js import three from pyodide.ffi import to_js document.getElementById('loader-container').style.display = 'none'; # Create the Viewer3D parameters = dict( show_joint_positions=True, show_axes=True, ) viewer3D = Viewer3D( document.getElementById('viewer3d'), parameters=parameters ) # Load the scene viewer3D.loadScene('/scenes/panda_box.xml') # Retrieve the robot robot = viewer3D.createRobot('panda', configs.Panda.new()) # Start the physics simulation viewer3D.physicsSimulatorPaused = False # Add some targets in layer 'NoShadows' viewer3D.activateLayer(Layers.NoShadows); viewer3D.addTarget('target1', [0.0, -0.4, 0.5], [-0.707, 0.707, 0.0, 0.0]) viewer3D.addTarget('target2', [0.6, 0.0, 0.5], [0.0, 1.0, 0.0, 0.0], '#cccc00', Shapes.Cone) # Add a big point at origin on layer 'Top' viewer3D.activateLayer(Layers.Top); viewer3D.addPoint( 'point1', [0.0, 0.0, 0.0], radius=0.05, color='#ff0000', shading=True ) # Add an Outline pass size = three.Vector2.new() viewer3D.renderer.getSize(size) outlinePass = OutlinePass.new(size, viewer3D.scene, viewer3D.camera, viewer3D.transformControls, to_js(robot.meshes)) outlinePass.visibleEdgeColor = three.Color.new(0, 1, 0) outlinePass.hiddenEdgeColor = three.Color.new(0, 1, 0) outlinePass.edgeStrength = 5.0 outlinePass.edgeThickness = 4.0 viewer3D.addPassBefore(outlinePass, Passes.TopRenderPass)
