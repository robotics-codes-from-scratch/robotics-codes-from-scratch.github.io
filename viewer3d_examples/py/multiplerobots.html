Multiple robots example (Python version)

This example shows how to use several robots in the same scene.

from viewer3d import Viewer3D, configs from js import document document.getElementById('loader-container').style.display = 'none'; # Create the Viewer3D viewer3D = Viewer3D(document.getElementById('viewer3d')) # Load the scene and retrieve the robots viewer3D.loadScene('/scenes/three_pandas.xml') robot1 = viewer3D.createRobot('panda1', configs.Panda.new(), 'panda1_') robot2 = viewer3D.createRobot('panda2', configs.Panda.new(), 'panda2_') robot3 = viewer3D.createRobot('panda3', configs.PandaNoHand.new(), 'panda3_') # Start the physics simulation viewer3D.physicsSimulatorPaused = False
