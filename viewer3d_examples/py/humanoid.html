Humanoid example (Python version)

This example uses a humanoid robot, with multiple end-effectors.


from viewer3d import Viewer3D, configs from js import document from pyodide.ffi import create_proxy document.getElementById('loader-container').style.display = 'none'; # Function to load the scene selected by the user def loadSelectedScene(evt=None): # Load the scene and retrieve the robot (if any) viewer3D.loadScene('/scenes/' + selector.value) robot = None if selector.value == 'g1_upperbody.xml': robot = viewer3D.createRobot('g1', configs.G1UpperBody.new()) elif selector.value == 'g1.xml': robot = viewer3D.createRobot('g1', configs.G1.new()) elif selector.value == 'g1_with_hands_upperbody.xml': robot = viewer3D.createRobot('g1', configs.G1WithHandsUpperBody.new()) elif selector.value == 'g1_with_hands.xml': robot = viewer3D.createRobot('g1', configs.G1WithHands.new()) # Start the physics simulation viewer3D.physicsSimulatorPaused = False # Create the Viewer3D viewer3D = Viewer3D(document.getElementById('viewer3d')) # Event handling to change the scene when the user selects one selector = document.getElementById('scenes') selector.addEventListener('change', create_proxy(loadSelectedScene)) loadSelectedScene()
